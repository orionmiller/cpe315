\section{Performance}
\[
  Performance = IPC \times CLK_{Rate} \times \frac{1}{I}
\]

\subsection{Ahmdal's Law}
\[
  Speedup = \frac{1}{\left(1 - F\right) + \frac{F}{s}}
\]

\subsection{Runtime}
\[
  Runtime = CPI \times CCT \times I
\]
